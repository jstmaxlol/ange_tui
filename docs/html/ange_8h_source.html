77static ange_input_t angeReadKey(void) {
78 char c;
79 ange_input_t input = { ANGE_KEY_UNKNOWN, 0 };
80
81 if (read(STDIN_FILENO, &c, 1) != 1) return input;
82
83 if (c == '\033') {
84 char seq[3] = {0};
85
86 fd_set set;
87 struct timeval timeout;
88
89 FD_ZERO(&set);
90 FD_SET(STDIN_FILENO, &set);
91
92 timeout.tv_sec = 0;
93 timeout.tv_usec = 50000;
94
95 int rv = select(STDIN_FILENO + 1, &set, NULL, NULL, &timeout);
96 if (rv == 0) {
97 input.type = ANGE_KEY_ESCAPE;
98 input.ch = 27;
99 return input;
100 }
101
102 if (read(STDIN_FILENO, &seq[0], 1) != 1) {
103 input.type = ANGE_KEY_ESCAPE;
104 input.ch = 27;
105 return input;
106 }
107 if (read(STDIN_FILENO, &seq[1], 1) != 1) {
108 input.type = ANGE_KEY_ESCAPE;
109 input.ch = 27;
110 return input;
111 }
112
113 if (seq[0] == '[') {
114 if (seq[1] >= '0' && seq[1] <= '9') {
115 if (read(STDIN_FILENO, &seq[2], 1) != 1) return input;
116 } else {
117 switch (seq[1]) {
118 case 'A': input.type = ANGE_KEY_ARROW_UP; break;
119 case 'B': input.type = ANGE_KEY_ARROW_DOWN; break;
120 case 'C': input.type = ANGE_KEY_ARROW_RIGHT; break;
121 case 'D': input.type = ANGE_KEY_ARROW_LEFT; break;
122 default: break;
123 }
124 }
125 }
126
127 return input;
128 } else if (c == '\177') {
129 input.type = ANGE_KEY_BACKSPACE;
130 input.ch = 127;
131 return input;
132 } else if (c == '\n' || c == '\r') {
133 input.type = ANGE_KEY_ENTER;
134 return input;
135 } else if (c >= 32 && c <= 126) {
136 input.type = ANGE_KEY_CHAR;
137 input.ch = c;
138 return input;
139 }
140
141 return input;
142}

155static void angeInit(void) {
156 if (ange_initialized) return;
157
158 tcgetattr(STDIN_FILENO, &tui_orig_termios);
159
160 struct termios raw = tui_orig_termios;
161
162 raw.c_lflag &= ~(ECHO | ICANON);
163 raw.c_cc[VMIN] = 1;
164 raw.c_cc[VTIME] = 0;
165
166 tcsetattr(STDIN_FILENO, TCSANOW, &raw);
167
168 printf("\033[?25l\033[2J\033[H");
169 fflush(stdout);
170
171 ange_initialized = 1;
172}

255static void angePrintP(const char* text, ange_align_t align) {
256 if (!ange_initialized) angeInit();
257
258 int len = strlen(text);
259 int pad = 0;
260
261 switch (align) {
262 case ANGE_ALIGN_CENTER: pad = (getColSize() - len) / 2; break;
263 case ANGE_ALIGN_RIGHT: pad = getColSize() - len; break;
264 case ANGE_ALIGN_LEFT: default: pad = 0;
265 }
266
267 if (pad < 0) pad = 0;
268 printf("\n%*s%s\n", pad, "", text);
269 fflush(stdout);
270}

352static int angeInputButton(ange_button_t* buttons, int count) {
353 int current = 0;
354
355 while (1) {
356 angeDrawButton(buttons, count, current);
357
358 ange_input_t input = angeReadKey();
359 switch (input.type) {
360 case ANGE_KEY_ARROW_UP:
361 current = (current - 1 + count) % count;
362 break;
363 case ANGE_KEY_ARROW_DOWN:
364 current = (current + 1) % count;
365 break;
366 case ANGE_KEY_ENTER:
367 return current;
368 default:
369 break;
370 }
371 }

383static int angeInputRadio(ange_radio_group_t* group) {
384 int current = group->selected_index;
385
386 while (1) {
387 printf("\033[H");
388 for (int i = 0; i < group->count; i++) {
389 const char* marker = (i == group->selected_index) ? "(o)" : "( )";
390 if (i == current)
391 printf("➤ %s %s\n", marker, group->options[i].label);
392 else
393 printf(" %s %s\n", marker, group->options[i].label);
394 }
395 fflush(stdout);
396
397 ange_input_t input = angeReadKey();
398 switch (input.type) {
399 case ANGE_KEY_ARROW_UP:
400 current = (current - 1 + group->count) % group->count;
401 break;
402 case ANGE_KEY_ARROW_DOWN:
403 current = (current + 1) % group->count;
404 break;
405 case ANGE_KEY_ENTER:
406 group->selected_index = current;
407 return current;
408 case ANGE_KEY_ESCAPE:
409 return -1;
410 default:
411 break;
412 }
413 }

425static void angeDrawTextbox(char* buffer, int max_len) {
426 int len = 0;
427 buffer[0] = '\0';
428
429 while (1) {
430 printf("\033[H> %s ", buffer);
431 fflush(stdout);
432
433 ange_input_t input = angeReadKey();
434 if (input.type == ANGE_KEY_BACKSPACE && len > 0) {
435 len--;
436 buffer[len] = '\0';
437 } else if (input.type == ANGE_KEY_CHAR && len < max_len - 1) {
438 buffer[len++] = input.ch;
439 buffer[len] = '\0';
440 } else if (input.type == ANGE_KEY_ENTER) {
441 break;
442 }
443 }
444}

490static int angeDrawNavbar(ange_navitem_t* items, int count, int selected_index) {
491 while (1) {
492 printf("\033[H");
493 for (int i = 0; i < count; i++) {
494 if (i == selected_index)
495 printf("[\033[7m%s\033[0m] ", items[i].label);
496 else
497 printf("[%s] ", items[i].label);
498 }
499 printf("\n");
500 fflush(stdout);
501
502 ange_input_t input = angeReadKey();
503
504 if (input.type == ANGE_KEY_ARROW_LEFT) {
505 selected_index = (selected_index - 1 + count) % count;
506 } else if (input.type == ANGE_KEY_ARROW_RIGHT) {
507 selected_index = (selected_index + 1) % count;
508 } else if (input.type == ANGE_KEY_ENTER) {
509 return selected_index;
510 } else if (input.type == ANGE_KEY_CHAR) {
511 for (int i = 0; i < count; i++) {
512 if (items[i].hotkey == input.ch || items[i].hotkey == input.ch + 32) {
513 return i;
514 }
515 }
516 }
517 }
