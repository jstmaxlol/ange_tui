69static ange_input_t angeReadKey(void) {
70 char c;
71 ange_input_t input = { ANGE_KEY_UNKNOWN, 0 };
72
73 if (read(STDIN_FILENO, &c, 1) != 1) return input;
74
75 if (c == '\033') {
76 char seq[2];
77 if (read(STDIN_FILENO, &seq[0], 1) != 1) return (ange_input_t){ ANGE_KEY_ESCAPE, 0 };
78 if (read(STDIN_FILENO, &seq[1], 1) != 1) return input;
79
80 if (seq[0] == '[') {
81 switch (seq[1]) {
82 case 'A': input.type = ANGE_KEY_ARROW_UP; break;
83 case 'B': input.type = ANGE_KEY_ARROW_DOWN; break;
84 case 'C': input.type = ANGE_KEY_ARROW_RIGHT; break;
85 case 'D': input.type = ANGE_KEY_ARROW_LEFT; break;
86 default: break;
87 }
88 }
89 } else if (c == '\n' || c == '\r') {
90 input.type = ANGE_KEY_ENTER;
91 } else if (c >= 32 && c <= 126) {
92 input.type = ANGE_KEY_CHAR;
93 input.ch = c;
94 }
95
96 return input;
97}

110static void angeInit(void) {
111 if (ange_initialized) return;
112
113 tcgetattr(STDIN_FILENO, &tui_orig_termios);
114
115 struct termios raw = tui_orig_termios;
116
117 raw.c_lflag &= ~(ECHO | ICANON);
118 raw.c_cc[VMIN] = 1;
119 raw.c_cc[VTIME] = 0;
120
121 tcsetattr(STDIN_FILENO, TCSANOW, &raw);
122
123 printf("\033[?25l\033[2J\033[H");
124 fflush(stdout);
125
126 ange_initialized = 1;
127}

217static void angePrintP(const char* text, ange_align_t align, int term_width) {
218 int len = strlen(text);
219 int pad = 0;
220
221 switch (align) {
222 case ANGE_ALIGN_CENTER: pad = (term_width - len) / 2; break;
223 case ANGE_ALIGN_RIGHT: pad = term_width - len; break;
224 case ANGE_ALIGN_LEFT: default: pad = 0;
225 }
226
227 if (pad < 0) pad = 0;
228 printf("\033[H%*s%s\n", pad, "", text);
229 fflush(stdout);
230}

405static void angeDrawTextbox(char* buffer, int max_len) {
406 int len = 0;
407 buffer[0] = '\0';
408
409 while (1) {
410 printf("\033[H> %s", buffer);
411 fflush(stdout);
412
413 ange_input_t input = angeReadKey();
414
415 if (input.type == ANGE_KEY_ENTER) {
416 break;
417 } else if (input.type == ANGE_KEY_CHAR && len < max_len - 1) {
418 buffer[len++] = input.ch;
419 buffer[len] = '\0';
420 } else if (input.type == ANGE_KEY_CHAR && input.ch == 127 && len > 0) {
421 len--;
422 buffer[len] = '\0';
423 }
424 }
